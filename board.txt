CREATE TABLE member (
id VARCHAR(20),
pass VARCHAR(20),
name VARCHAR(20),
phone VARCHAR(20),
team VARCHAR(20),
emai VARCHAR(40),
postcode VARCHAR(20),
address VARCHAR(40),
date datetime
);

CREATE TABLE board(
b_num INT,
id VARCHAR(20),
subject VARCHAR(40),
content VARCHAR(500),
readcount INT,
b_date datetime,
pass varchar(20)
);

ALTER TABLE member ADD CONSTRAINT PK_MEMBER PRIMARY KEY (id);

ALTER TABLE board ADD CONSTRAINT PK_BOARD PRIMARY KEY (b_num);
